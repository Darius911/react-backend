CREATE TABLE visits (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    pets_name VARCHAR(255) NOT NULL,
	owner_name VARCHAR(255) NOT NULL,
	date DATE NOT NULL,
	time TIME NOT NULL,
	notes VARCHAR (255)
    
);

CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL,
	role VARCHAR(100)
    );
